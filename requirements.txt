# NumPy - работа с массивами чисел (математика, обработка сигналов)
# Используется для: анализа аудио данных, вычисления корреляций
numpy

# SciPy - научные вычисления (обработка сигналов, FFT)
# Используется для: извлечения характеристик звука, фильтрации
scipy

# PyAudio - запись звука с микрофона
# Используется для: захвата аудио в реальном времени
# 
# ВНИМАНИЕ: На Windows может потребоваться установка через:
# pip install pipwin
# pipwin install pyaudio
pyaudio

# Pillow - работа с изображениями (для создания иконок)
# Используется только для генерации beer.ico и setup.ico
Pillow

# PyInstaller - создание .exe файлов
# Используется для: сборки программы в исполняемый файл
# Команда сборки:
# pyinstaller --onefile --windowed --name=SkufUp --icon=beer.ico gui_app.py
pyinstaller
